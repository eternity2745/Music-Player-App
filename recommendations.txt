select s.title, s.author, s.language, s.genre from songs s, listening_history lh WHERE s.id NOT IN (SELECT DISTINCT song_id FROM listening_history WHERE username = 'The King') AND (s.genre IN (SELECT DISTINCT genre FROM listening_history WHERE username = 'The King') OR s.language IN (SELECT DISTINCT language FROM listening_history WHERE username = 'The King') OR s.author IN (SELECT DISTINCT author FROM listening_history WHERE username = 'The King')) GROUP BY s.title ORDER BY RAND() LIMIT 8;
